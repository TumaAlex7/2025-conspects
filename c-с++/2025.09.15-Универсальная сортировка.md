void qsort(void *array, size_t n, size_t elem_size, int (*cmp)(void *, void *)){

void swap(char *sa, char *sb, size_t elem_size){
    int i = 0;
    for (i = 0; i < elem_size; i++){
        char tmp = *(sa + i);
        *(sa + i) = *(sb + i);
        *(sb + i) = tmp;
    }
}

int int_cmp(void *a, void *b){
    return *(int *)a - *(int *)b;
}
